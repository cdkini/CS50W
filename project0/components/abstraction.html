            // Parent or superclass
            public class Animal {

                private String scientificName;
                private float weight;
                private String noise;


                public Animal(String scientificName, float weight, String noise) {
                    this.classification = scientificName;
                    this.weight = weight;
                    this.noise = noise;
                }

                public void makeNoise() {
                    System.out.println(noise);
                }
            }

            // Child or subclass
            public class Dog extends Animal {
                
                private String breed;

                public Dog(String breed, float weight) {
                    super("Canis lupis familiaris", weight, "bark!");
                    this.breed = breed;
                }

            }
